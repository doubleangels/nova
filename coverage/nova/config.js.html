function isSet(value) {
  return value != null && String(value).trim() !== '';
}
 
const missing = REQUIRED_ENV_VARS.filter(name => !isSet(process.env[name]));
if (missing.length > 0) {
  console.error('Missing required environment variable(s). Bot cannot start.');
